// Variável global para rastrear a posição X do carro let carX = 175; // Começa no centro (400/2 - 25) const player = document.getElementById('player'); const container = document.querySelector('.game-container'); const speed = 20; // Quantos pixels o carro anda por clique // 1. Movimento via Mouse (Mantido) function moveCar(e) { const rect = container.getBoundingClientRect(); carX = e.clientX - rect.left - 25; updateCarPosition(); } // 2. Movimento via Teclado window.addEventListener('keydown', (e) => { if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') { carX -= speed; } else if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') { carX += speed; } updateCarPosition(); }); // 3. Função Única para Atualizar e Validar Limites function updateCarPosition() { // Impede que o carro saia da pista (0 a 350px) if (carX < 0) carX = 0; if (carX > 350) carX = 350; player.style.left = carX + 'px'; player.style.transform = 'translateX(0)'; }
